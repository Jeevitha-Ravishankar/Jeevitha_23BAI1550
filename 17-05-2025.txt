CREATE DATABASE college;
USE college;
CREATE TABLE student_details (
    student_id INT ,
    name VARCHAR(50),
    marks INT,
    email VARCHAR(100)
);

CREATE TABLE faculty_details (
    faculty_id INT ,
    name VARCHAR(50),
    subject VARCHAR(50),
    phone VARCHAR(15)
);

CREATE TABLE course_details (
    course_id INT ,
    course_name VARCHAR(100),
    credits INT,
    department VARCHAR(50)
);
